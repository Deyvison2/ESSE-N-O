<script>
    function executarCalculo() {
        // Simula o JOptionPane.showInputDialog
        let input1 = prompt("Digite o primeiro valor:");
        let input2 = prompt("Digite o segundo valor:");
        let input3 = prompt("Digite a velocidade da norea:");

        // Converte para Inteiro (Integer.parseInt)
        let valor1 = parseInt(input1);
        let valor2 = parseInt(input2);
        let velocidadeNorea = parseInt(input3);

        // Validação básica
        if (isNaN(valor1) || isNaN(valor2) || isNaN(velocidadeNorea)) {
            alert("Erro: Digite apenas números inteiros!");
            return;
        }
        
        if (velocidadeNorea === 0) {
            alert("Erro: A velocidade não pode ser zero!");
            return;
        }

        // Lógica idêntica ao seu código Java
        let resultado = valor2 - valor1;
        let vazao = resultado / velocidadeNorea;

        // Formatação com 3 casas decimais (%.3f) e exibe o JOptionPane.showMessageDialog
        let mensagem = "A vazão é: " + vazao.toFixed(3).replace('.', ',');
        alert(mensagem);
    }
</script>
